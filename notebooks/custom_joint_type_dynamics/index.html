    Custom joint type dynamics

    This example is available as a jupyter notebook here.

    Defining a custom Joint Type that supports dynamical simulation¤

    In this notebook we will define a new joint type that is a hinge joint with a random joint axes direction.

    It will also support dynamical simulation.

    import x_xy

import jax
import jax.numpy as jnp

import mediapy as media

    We will give this new joint type the identifier rr (random revolute).

    from x_xy.algorithms.jcalc import _draw_rxyz

# we use this `params` input to specify the joint-axes, if we later then randomize the attribute of the system object
# we will have the effect of a hinge joint with a randomized joint axes direction

# set the dimensionality of the attribute to 3 for each joint / link in the system.
# This is save because the library itself does not use this attribute at all
x_xy.base.update_n_joint_params(3)

def _rr_transform(q, params):
    axis = params
    q = jnp.squeeze(q)
    rot = x_xy.maths.quat_rot_axis(axis, q)
    return x_xy.Transform.create(rot=rot)

def _motion_fn(params):
    return x_xy.base.Motion.create(ang=params)

# here we do specify the motion vectors
rr_joint = x_xy.JointModel(_rr_transform, motion=[_motion_fn], rcmg_draw_fn=_draw_rxyz)

# we register the joint; Note that qd_width is not zero now, but rather one
x_xy.register_new_joint_type("rr", rr_joint, q_width=1, qd_width=1)

    xml_str = """
<x_xy>
    <options gravity="0 0 9.81" dt="0.01"/>
    <worldbody>
        <geom type="xyz" dim="0.1"/>
        <body name="pendulum" joint="rr" pos="0 0 0.5" damping=".01">
            <geom type="xyz" dim="0.1"/>
            <geom type="box" mass="0.5" pos="0.25 0 0" dim="0.5 0.1 0.1"/>
        </body>
    </worldbody>
</x_xy>
"""

sys = x_xy.load_sys_from_str(xml_str)
sys = sys.replace(links=sys.links.replace(joint_params = x_xy.maths.safe_normalize(jnp.array([[0, 1.0, -1.0]]))))

    state = x_xy.State.create(sys)
xs = []
for t in range(500):
    state = jax.jit(x_xy.step)(sys, state)
    xs.append(state.x)

    def show_video(sys, xs: x_xy.Transform):
    frames = x_xy.render(sys, [xs[i] for i in range(0, len(xs), 4)], camera=None)
    media.show_video(frames, fps=25)

show_video(sys, xs)
